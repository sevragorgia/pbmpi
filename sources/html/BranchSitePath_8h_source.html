92  static int rrindex(int i, int j, int nstate) {
93  return (i<j) ? (2 * nstate - i - 1) * i / 2 + j - i - 1 : (2 * nstate - j - 1) * j / 2 + i - j - 1 ;

96  void AddRateSuffStat(int& count, double& beta, double factor, const double* rr, const double* stat, int nstate) {
97  Plink* link = Init();
98  while (link) {
99  int state = link->GetState();
100  double tmp = 0;
101  for (int k=0; k<nstate; k++) {
102  if (k != state) {
103  tmp += rr[rrindex(state,k,nstate)] * stat[k];
104  }
105  }
106  beta += GetRelativeTime(link) * factor * tmp;
107  if (link != last) {
108  count ++;
109  }
110  link = link->Next();
111  }
112  }

146  void AddGeneralPathRateSuffStat(int& count, double& beta, double factor, SubMatrix* mat) {
147  Plink* link = Init();

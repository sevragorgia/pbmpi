306 
307  chronocollapse.Start();

311  GammaBranchProcess::Move(0.1 * tuning,10);
312  GammaBranchProcess::Move(tuning,10);

314 
315  GlobalUpdateParameters();
316  AACodonMutSelSBDPProfileProcess::Move(tuning,1,15);
317  chronosuffstat.Stop();

321  chronounfold.Stop();

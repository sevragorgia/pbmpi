220  virtual double Move(double tuning = 1.0) {
221  chronototal.Start();
222  propchrono.Start();
223  if (! fixbl) {
224  BranchLengthMove(tuning);
225  BranchLengthMove(0.1 * tuning);
226  }
227  if (! fixtopo) {
228  MoveTopo(NSPR,NNNI);
229  }
230  propchrono.Stop();
231 
232 
233  for (int rep=0; rep<5; rep++) {
234  GlobalCollapse();
235 
236  if (! fixbl) {
237  GammaBranchProcess::Move(tuning,10);
238  GammaBranchProcess::Move(0.1*tuning,10);
239  }
240 
241  GlobalUpdateParameters();
242  DGamRateProcess::Move(tuning,10);
243  DGamRateProcess::Move(0.3*tuning,10);
244  DGamRateProcess::Move(0.03*tuning,10);
245 
246  PoissonSBDPProfileProcess::Move(1,1,1);
247  if (iscodon) {
248  PoissonSBDPProfileProcess::Move(0.1,1,3);
249  PoissonSBDPProfileProcess::Move(0.01,1,3);
250  }
251  GlobalUpdateParameters();
252 
253  GlobalUnfold();
254  }
255  chronototal.Stop();
256 
257  return 1;

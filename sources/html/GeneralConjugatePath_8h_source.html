204  cerr << GetMatrix()->GetNstate() << '\n';
205  exit(1);

208  int total = 0;
209  for (map<pair<int,int>, int>::iterator i = paircount.begin(); i!= paircount.end(); i++) {

219  incl = true;
220  }
221  else {
222  incl = false;
223  }
224  }

226  total += n;
227  }

265  double SuffStatLogProb() {
266  double total = 0;
267 
268  // root part
269  if (! GetTime()) {
270  const double* rootstat = GetStationary();
271  for (map<int,int>::iterator i = rootcount.begin(); i!= rootcount.end(); i++) {
272  total += i->second * log(rootstat[i->first]);
273  }
274  }
275 
276  // non root part
277  if (GetTime()) {
278  int totnsub = 0;
279  double totscalestat = 0;
280  SubMatrix& mat = *GetMatrix();
281  for (map<int,double>::iterator i = waitingtime.begin(); i!= waitingtime.end(); i++) {
282  totscalestat += i->second * mat(i->first,i->first);
283  }
284  for (map<pair<int,int>, int>::iterator i = paircount.begin(); i!= paircount.end(); i++) {
285  total += i->second * log(mat(i->first.first, i->first.second));
286  totnsub += i->second;
287  }
288  total += GetRate() * GetTime() * totscalestat;
289  if (totnsub) {
290  total += totnsub * log(GetRate() * GetTime());
291  }
292  }
293  // return 0;
294  return total;
295  }

272  double Move(double tuning = 1.0) {
273 
274  chronototal.Start();
275 
276  propchrono.Start();
277  if (! fixbl) {
278  BranchLengthMove(tuning);
279  BranchLengthMove(0.1 * tuning);
280  }
281  if (! fixtopo) {
282  MoveTopo(NSPR,NNNI);
283  }
284  propchrono.Stop();
285 
286 
287  for (int rep=0; rep<5; rep++) {
288  GlobalCollapse();
289 
290  if (! fixbl) {
291  GammaBranchProcess::Move(tuning,10);
292  GammaBranchProcess::Move(0.1*tuning,10);
293  }
294 
295  GlobalUpdateParameters();
296  DGamRateProcess::Move(tuning,10);
297  DGamRateProcess::Move(0.3*tuning,10);
298  DGamRateProcess::Move(0.03*tuning,10);
299 
300  GlobalUpdateParameters();
301  ExpoConjugateGTRSBDPProfileProcess::Move(1,1,2);
302  if (iscodon){
303  ExpoConjugateGTRSBDPProfileProcess::Move(0.1,1,3);
304  ExpoConjugateGTRSBDPProfileProcess::Move(0.01,1,3);
305  }
306  GlobalUpdateParameters();
307 
308  if ((! fixrr) && (! fixbl)){
309  LengthRelRateMove(1,10);
310  LengthRelRateMove(0.1,10);
311  LengthRelRateMove(0.01,10);
312  }
313 
314  GlobalUnfold();
315  }
316 
317  chronototal.Stop();
318 
319  return 1;
320  }

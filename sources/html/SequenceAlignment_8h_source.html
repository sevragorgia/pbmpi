99  SequenceAlignment(SequenceAlignment* from, int N, int Ngene, int* genesize, int* exclude) {
100 
101  Ntaxa = from->Ntaxa;
102 
103  int* include = new int[Ngene];
104  int nex = 0;
105  for (int i=0; i<Ngene; i++) {
106  nex += exclude[i];
107  include[i] = exclude[i];
108  }
109 
110  int Nleft = Ngene - nex;
111  for (int i=0; i<N; i++) {
112  int gene = (int) (Nleft * rnd::GetRandom().Uniform());
113  int k = 0;
114  while (gene) {
115  if (!include[k]) {
116  gene--;
117  }
118  k++;
119  }
120  if (include[k]) {
121  k++;
122  }
123 
124  if (k >= Ngene) {
125  cerr << "error : Ngene overflow\n";
126  exit(1);
127  }
128 
129  include[k] = 1;
130  Nleft--;
131  }
132 
133  int Ncheck = 0;
134  int totsize = 0;
135  for (int i=0; i<Ngene; i++) {
136  include[i] -= exclude[i];
137  if (include[i]) {
138  Ncheck++;
139  totsize += genesize[i];
140  }
141  }
142 
143  Nsite = totsize;
144 
145  taxset = from->taxset;
146  statespace = from->statespace;
147 
148  Data = new int*[Ntaxa];
149  for (int i=0; i<Ntaxa; i++) {

151  }
152

651  double CompositionalHeterogeneity(double* taxstat, ostream* os, double& meandist) {
652 
653  int Nstate = GetNstate();
654  double** taxfreq = new double*[Ntaxa];
655  for (int j=0; j<Ntaxa; j++) {
656  taxfreq[j] = new double[Nstate];
657  for (int k=0; k<Nstate; k++) {
658  taxfreq[j][k] = 0;
659  }
660  }
661 
662  for (int i=0; i<Nsite; i++) {
663  for (int j=0; j<Ntaxa; j++) {
664  int state = GetState(j,i);
665  if (state != unknown) {
666  taxfreq[j][state]++;
667  }
668  }
669  }
670 
671  for (int j=0; j<Ntaxa; j++) {
672  double total = 0;
673  for (int k=0; k<Nstate; k++) {
674  total += taxfreq[j][k];
675  }
676  for (int k=0; k<Nstate; k++) {
677  taxfreq[j][k] /= total;
678  }
679  }
680 
681  // make global freqs out of tax-specific freqs
682  double* globalfreq = new double[Nstate];
683  for (int k=0; k<Nstate; k++) {
684  globalfreq[k] = 0;
685  for (int j=0; j<Ntaxa; j++) {
686  globalfreq[k] += taxfreq[j][k];
687  }
688  globalfreq[k] /= Ntaxa;
689  }
690 
691  for (int j=0; j<Ntaxa; j++) {
692  for (int k=0; k<Nstate; k++) {
693  if (os) {
694  (*os) << taxfreq[j][k] << '\t';
695  }
696  }
697  if (os) {
698  (*os) << '\n';
699  }
700  }
701  if (os) {
702  (*os) << '\n';
703  }
704 
705  // compute max and mean squared distance
706  double maxdist = 0;
707  meandist = 0;
708  for (int j=0; j<Ntaxa; j++) {
709  double dist = 0;
710  for (int k=0; k<Nstate; k++) {
711  double tmp = (taxfreq[j][k] - globalfreq[k]);
712  dist += tmp * tmp;
713  }
714  taxstat[j] = dist;
715  if (maxdist < dist) {
716  maxdist = dist;
717  }
718  meandist += dist;
719  }
720  meandist /= Ntaxa;
721 
722  delete[] globalfreq;
723  for (int j=0; j<Ntaxa; j++) {
724  delete[] taxfreq[j];
725  }
726  delete[] taxfreq;
727 
728  return maxdist;
729  }

254 inline double SubMatrix::GetFiniteTimeTransitionProb(int stateup, int statedown, double efflength) {
255  double** invp = GetInvEigenVect();
256  double** p = GetEigenVect();
257  double* l = GetEigenVal();
258  double tot = 0;
259  for (int i=0; i<GetNstate(); i++) {
260  tot += p[stateup][i] * exp(efflength * l[i]) * invp[i][statedown];
261  }
262  return tot;
263 }

265 inline int SubMatrix::DrawUniformizedTransition(int state, int statedown, int n) {
266 
267  double* p = new double[GetNstate()];
268  double tot = 0;
269  for (int l=0; l<GetNstate(); l++) {
270  tot += Power(1,state,l) * Power(n,l,statedown);
271  p[l] = tot;
272  }
273 
274  double s = tot * rnd::GetRandom().Uniform();
275  int k = 0;
276  while ((k<GetNstate()) && (s > p[k])) {
277  k++;
278  }
279  if (k == GetNstate()) {
280  cerr << "error in DrawUniformizedTransition: overflow\n";
281  throw;
282  }
283  delete[] p;
284  return k;
285 }

288 inline int SubMatrix::DrawUniformizedSubstitutionNumber(int stateup, int statedown, double efflength) {
289 
290  double mu = GetUniformizationMu();
291  double fact = exp(- efflength * mu);
292  int m = 0;
293  double total = (stateup==statedown) * fact;
294  double Z = GetFiniteTimeTransitionProb(stateup,statedown,efflength);
295  double q = rnd::GetRandom().Uniform() * Z;
296 
297  while ((m<UniSubNmax) && (total < q)) {
298  m++;
299  fact *= mu * efflength / m;
300  total += Power(m,stateup,statedown) * fact;
301  if ((total-Z)>1e-12) {
302  cerr << "error in DrawUniformizedSubstitutionNumber: normalising constant\n";
303  cerr << total << '\t' << Z << '\n';
304  cerr << mu << '\n';
305  cerr << m << '\n';
306  cerr << stateup << '\t' << statedown << '\n';
307 
308  // ToStream(cerr);
309  // CheckReversibility();
310  throw;
311  }
312  }
313  if (m == UniSubNmax) {
314  nunisubcount++;
315  }
316  return m;
317 }

319 inline int SubMatrix::DrawOneStep(int state) {
320 
321  const double* row = GetRow(state);
322  double p = -row[state] * rnd::GetRandom().Uniform();
323  int k = -1;
324  double tot = 0;
325  do {
326  k++;
327  if (k != state) {
328  tot += row[k];
329  }
330  }
331  while ((k<GetNstate()) && (tot < p));
332  if (k == GetNstate()) {
333  cerr << "error in DrawOneStep\n";
334  cerr << GetNstate() << '\n';
335  for (int k=0; k<GetNstate(); k++) {
336  cerr << row[k] << '\n';
337  }
338  // ToStream(cerr);
339  exit(1);
340  }
341  return k;
342 }

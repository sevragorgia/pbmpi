94  int width = plaindata->GetNsite()/(nprocs-1);
95  insitemin = (myid-1)*width;
96  if (myid == (nprocs-1)) {
97  insitemax = plaindata->GetNsite();

99  else {
100  insitemax = myid*width;
101  }
102  }
103 
104  Create(tree,plaindata,nratecat,inrrtype,insitemin,insitemax);

203  double Move(double tuning = 1.0) {
204 
205  chronototal.Start();
206 
207  propchrono.Start();
208  if (! fixbl) {
209  BranchLengthMove(tuning);
210  BranchLengthMove(0.1 * tuning);
211  }
212  if (! fixtopo) {
213  MoveTopo(10,0);
214  }
215  propchrono.Stop();
216 
217  GlobalCollapse();
218 
219  if (! fixbl) {
220  GammaBranchProcess::Move(tuning,10);
221  }
222 
223  GlobalUpdateParameters();
224  DGamRateProcess::Move(0.3*tuning,10);
225  DGamRateProcess::Move(0.03*tuning,10);
226 
227  ExpoConjugateGTRDPProfileProcess::Move(1,1,10);
228 
229  if ((! fixrr) && (! fixbl)) {
230  LengthRelRateMove(1,10);
231  LengthRelRateMove(0.1,10);
232  LengthRelRateMove(0.01,10);
233  }
234 
235  GlobalUnfold();
236 
237  chronototal.Stop();
238 
239  return 1;
